ASM

可以使用 Tool godbolt 得到組譯結果。稍微修改一下就能嵌入 C 裡，特別小心作業系統架構，符合才能運行，同時也要避免編譯時期最佳化 -O2, -O1 ... 等造成的影響。

#include <stdio.h>
volatile void asm_solver(volatile int h, volatile int w, volatile int d) {
    volatile int area;
       volatile int volume;
    __asm__ __volatile__(
        /* (h+d)*w => edx */
        "movl    %1   , %%edx\n\t"
        "movl    %3   , %%eax\n\t"
        "addl    %%edx, %%eax\n\t"
        "imull    %2   , %%eax\n\t"
        "movl    %%eax, %%edx\n\t"
        /* h*d => eax, eax + edx => (h+d)*w+h*d*/
        "movl    %3   , %%eax\n\t"
        "imull    %1   , %%eax\n\t"
        "addl    %%edx, %%eax\n\t"
        /* ((h+d)*w+h*d) * 2 */
        "addl    %%eax, %%eax\n\t"
        "movl   %%eax, %0\n\t"
    : "=m"(area) 
    : "m"(h), "m"(w), "m"(d)
    : "memory");
 
    __asm__ __volatile__(
        "movl    %1   , %%eax\n\t"
        "imull  %2   , %%eax\n\t"
        "imull  %3   , %%eax\n\t"
        "movl   %%eax, %0"
    : "=m"(volume) 
    : "m"(h), "m"(w), "m"(d)
    : "memory");
    printf("%d\n%d\n", area, volume);
}
int main() {
    int h, w, d;
    scanf("%d %d %d",&h, &w, &d);
    asm_solver(h, w, d);
    return 0;
}
