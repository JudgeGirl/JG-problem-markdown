#include <stdio.h>
#include <stdlib.h>
#include <string.h>
 
int matrix[405][405];
int lakes[80000];
 
void swap(int *a, int *b) {
    int temp;
    temp = *a;
    *a = *b;
    *b = temp;
}
 
int lake(int i, int j) {
    int ans = 1;
    matrix[i][j] = 0;
    if ( matrix[i+1][j] ) ans += lake(i+1, j);
    if ( matrix[i][j+1] ) ans += lake(i, j+1);
    if ( matrix[i-1][j] ) ans += lake(i-1, j);
    if ( matrix[i][j-1] ) ans += lake(i, j-1);
    return ans;
}
 
int run(void) {
 
    int n, m, i, j;
    scanf("%d%d", &n, &m);
 
    for(j = 1; j <= n; j++)
        for(i = 1; i <= m; i++)
            scanf("%d", &matrix[i][j]);
 
    int num=0;
    for(j = 1; j <= n; j++)
        for(i = 1; i <= m; i++)
            if ( matrix[i][j] ) {
                lakes[num++] = lake(i, j);
            }
 
    for (i=0; i<num-1; i++)
        for (j=num-1; j>i; j--)
            if ( lakes[j] > lakes[i] ) swap(&lakes[i], &lakes[j]);
 
    for (i=0; i<num; i++) printf("%d\n", lakes[i]);
 
    return 0;
}
extern int main2(void) __asm__ ("main2");  
 
int main2() {  
    run();
    exit(0);  
}
